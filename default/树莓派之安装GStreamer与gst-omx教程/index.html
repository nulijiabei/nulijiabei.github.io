树莓派之安装GStreamer与gst-omx教程

// 在最新的 GStreamer 1.18.x 使用 meson 进行编译

// 下载 GStreamer 提供的 Build 工具
https://github.com/GStreamer/gst-build

// 下载 Meson 工具
https://github.com/mesonbuild/meson

// 安装 ninja build ...
https://github.com/martine/ninja.git
或：apt-get install ninja-build

// Build  ...
cd gst-build
../meson.py build
ninja -C build
ninja -C build/ install

// 改 gitlab.gnome.org 或 gitlab.freedesktop.org 为 github.com
../gst-build/subprojects/xxx.wrap 

// 默认扩展插件位置 ...
/usr/local/lib/aarch64-linux-gnu/gstreamer-1.0

// 查询扩展插件 ...
gst-inspect-1.0 --gst-plugin-path=/usr/local/lib/gstreamer-1.0 | grep mpp

sudo apt-get install -y build-essential autotools-dev automake autoconf \

cd $HOME
mkdir packages
cd packages/
mkdir gstreamer-1.10
cd gstreamer-1.10


git clone https://github.com/GStreamer/gstreamer
cd gstreamer
git checkout -t origin/1.10
./autogen.sh
make
sudo make install
cd ..


git clone https://github.com/GStreamer/gst-plugins-base
cd gst-plugins-base
git checkout -t origin/1.10
./autogen.sh
make
sudo make install

git clone https://github.com/GStreamer/gst-plugins-bad

# Install libusb-1.0 to enable  uvch264src
sudo apt-get install libusb-1.0

cd gst-plugins-bad
git checkout -t origin/1.10
export LD_LIBRARY_PATH=/usr/local/lib/ path
LDFLAGS='-L/opt/vc/lib' CPPFLAGS='-I/opt/vc/include -I/opt/vc/include/interface/vcos/pthreads -I/opt/vc/include/interface/vmcs_host/linux' ./autogen.sh
make CFLAGS+="-Wno-error"
sudo make install
cd ..

################# COMPILE GST-OMX ##########


git clone https://github.com/GStreamer/gst-omx
cd gst-omx
git checkout -t origin/1.10
LDFLAGS='-L/opt/vc/lib' CPPFLAGS='-I/opt/vc/include -I/opt/vc/include/IL -I/opt/vc/include/interface/vcos/pthreads -I/opt/vc/include/interface/vmcs_host/linux' ./autogen.sh --with-omx-target=rpi
make CFLAGS+="-Wno-error"
sudo make install
